00131   unsigned n = vec.size();
00132   std::vector<unsigned> idx(n, 0);
00133   std::vector<std::vector<unsigned>> result;
00134   // if any one of the elements is zero there are no
00135   // combinations to build
00136   if (std::find(vec.begin(), vec.end(), 0) != vec.end()) {
00137     return result;
00138   }
00139   result.push_back(idx);
00140   bool exit_loop = false;
00141   while (exit_loop == false) {
00142     // Find the first index we can increment (if possible)
00143     for (unsigned i = 0; i < n; ++i) {
00144       if ((idx[i] + 1) == vec[i]) {
00145         if (i != n - 1) {
00146           idx[i] = 0;
00147         } else {
00148           // we're done
00149           exit_loop = true;
00150           break;
00151         }
00152       } else {
00153         ++(idx[i]);
00154         result.push_back(idx);
00155         break;
00156       }
00157     }
00158   }
