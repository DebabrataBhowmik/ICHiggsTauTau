00411     std::vector<HistMapping> const& mappings) {

00413   for (unsigned a = 0; a < attempts.size(); ++a) {
00414     for (unsigned m = 0; m < mappings.size(); ++m) {
00415       if ((attempts[a].first == mappings[m].process) &&
00416         (attempts[a].second == mappings[m].category)) {
