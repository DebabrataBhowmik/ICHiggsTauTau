00098 // Bitwise right rotate.  Normally this will compile to a single
00099 // instruction, especially if the shift is a manifest constant.
00100 static uint64 Rotate(uint64 val, int shift) {
00101   // Avoid shifting by 64: doing so yields an undefined result.
00102   return shift == 0 ? val : ((val >> shift) | (val << (64 - shift)));
00103 }

00105 // Equivalent to Rotate(), but requires the second arg to be non-zero.
00106 // On x86-64, and probably others, it's possible for this to compile
00107 // to a single instruction if both args are already in registers.
00108 static uint64 RotateByAtLeast1(uint64 val, int shift) {
00109   return (val >> shift) | (val << (64 - shift));
00110 }

00120 static uint64 HashLen0to16(const char *s, size_t len) {
00121   if (len > 8) {
00122     uint64 a = Fetch64(s);
00123     uint64 b = Fetch64(s + len - 8);
00124     return HashLen16(a, RotateByAtLeast1(b + len, len)) ^ b;
00125   }
00126   if (len >= 4) {
00127     uint64 a = Fetch32(s);
00128     return HashLen16(len + (a << 3), Fetch32(s + len - 4));
00129   }
00130   if (len > 0) {
00131     uint8 a = s[0];
00132     uint8 b = s[len >> 1];
00133     uint8 c = s[len - 1];
00134     uint32 y = static_cast<uint32>(a) + (static_cast<uint32>(b) << 8);
00135     uint32 z = len + (static_cast<uint32>(c) << 2);
00136     return ShiftMix(y * k2 ^ z * k3) * k2;
00137   }
00138   return k2;
00139 }

00141 // This probably works well for 16-byte strings as well, but it may be overkill
00142 // in that case.
00143 static uint64 HashLen17to32(const char *s, size_t len) {
00144   uint64 a = Fetch64(s) * k1;
00145   uint64 b = Fetch64(s + 8);
00146   uint64 c = Fetch64(s + len - 8) * k2;
00147   uint64 d = Fetch64(s + len - 16) * k0;
00148   return HashLen16(Rotate(a - b, 43) + Rotate(c, 30) + d,
00149                    a + Rotate(b ^ k3, 20) - c + len);
00150 }

00152 // Return a 16-byte hash for 48 bytes.  Quick and dirty.
00153 // Callers do best to use "random-looking" values for a and b.
00154 static pair<uint64, uint64> WeakHashLen32WithSeeds(
00155     uint64 w, uint64 x, uint64 y, uint64 z, uint64 a, uint64 b) {
00156   a += w;
00157   b = Rotate(b + a + z, 21);
00158   uint64 c = a;
00159   a += x;
00160   a += y;
00161   b += Rotate(a, 44);
00162   return make_pair(a + z, b + c);

00176 // Return an 8-byte hash for 33 to 64 bytes.
00177 static uint64 HashLen33to64(const char *s, size_t len) {
00178   uint64 z = Fetch64(s + 24);
00179   uint64 a = Fetch64(s) + (len + Fetch64(s + len - 16)) * k0;
00180   uint64 b = Rotate(a + z, 52);
00181   uint64 c = Rotate(a, 37);
00182   a += Fetch64(s + 8);
00183   c += Rotate(a, 7);
00184   a += Fetch64(s + 16);
00185   uint64 vf = a + z;
00186   uint64 vs = b + Rotate(a, 31) + c;
00187   a = Fetch64(s + 16) + Fetch64(s + len - 32);
00188   z = Fetch64(s + len - 8);
00189   b = Rotate(a + z, 52);
00190   c = Rotate(a, 37);
00191   a += Fetch64(s + len - 24);
00192   c += Rotate(a, 7);
00193   a += Fetch64(s + len - 16);
00194   uint64 wf = a + z;
00195   uint64 ws = b + Rotate(a, 31) + c;
00196   uint64 r = ShiftMix((vf + ws) * k2 + (wf + vs) * k0);
00197   return ShiftMix(r * k0 + vs) * k2;
00198 }

00200 uint64 CityHash64(const char *s, size_t len) {
00201   if (len <= 32) {
00202     if (len <= 16) {
00203       return HashLen0to16(s, len);
00204     } else {
00205       return HashLen17to32(s, len);
00206     }
00207   } else if (len <= 64) {
00208     return HashLen33to64(s, len);
00209   }
00210 
00211   // For strings over 64 bytes we hash the end first, and then as we
00212   // loop we keep 56 bytes of state: v, w, x, y, and z.
00213   uint64 x = Fetch64(s + len - 40);
00214   uint64 y = Fetch64(s + len - 16) + Fetch64(s + len - 56);
00215   uint64 z = HashLen16(Fetch64(s + len - 48) + len, Fetch64(s + len - 24));
00216   pair<uint64, uint64> v = WeakHashLen32WithSeeds(s + len - 64, len, z);
00217   pair<uint64, uint64> w = WeakHashLen32WithSeeds(s + len - 32, y + k1, x);
00218   x = x * k1 + Fetch64(s);
00219 
00220   // Decrease len to the nearest multiple of 64, and operate on 64-byte chunks.
00221   len = (len - 1) & ~static_cast<size_t>(63);
00222   do {
00223     x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
00224     y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
00225     x ^= w.second;
00226     y += v.first + Fetch64(s + 40);
00227     z = Rotate(z + w.first, 33) * k1;
00228     v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
00229     w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
00230     std::swap(z, x);
00231     s += 64;
00232     len -= 64;
00233   } while (len != 0);
00234   return HashLen16(HashLen16(v.first, w.first) + ShiftMix(y) * k1 + z,
00235                    HashLen16(v.second, w.second) + x);
00236 }

00251 // A subroutine for CityHash128().  Returns a decent 128-bit hash for strings
00252 // of any length representable in signed long.  Based on City and Murmur.
00253 static uint128 CityMurmur(const char *s, size_t len, uint128 seed) {
00254   uint64 a = Uint128Low64(seed);
00255   uint64 b = Uint128High64(seed);
00256   uint64 c = 0;
00257   uint64 d = 0;
00258   signed long l = len - 16;
00259   if (l <= 0) {  // len <= 16
00260     a = ShiftMix(a * k1) * k1;
00261     c = b * k1 + HashLen0to16(s, len);
00262     d = ShiftMix(a + (len >= 8 ? Fetch64(s) : c));
00263   } else {  // len > 16
00264     c = HashLen16(Fetch64(s + len - 8) + k1, a);
00265     d = HashLen16(b + len, c + Fetch64(s + len - 16));
00266     a += d;
00267     do {
00268       a ^= ShiftMix(Fetch64(s) * k1) * k1;
00269       a *= k1;
00270       b ^= a;
00271       c ^= ShiftMix(Fetch64(s + 8) * k1) * k1;
00272       c *= k1;
00273       d ^= c;
00274       s += 16;
00275       l -= 16;
00276     } while (l > 0);
00277   }
00278   a = HashLen16(a, c);
00279   b = HashLen16(d, b);
00280   return uint128(a ^ b, HashLen16(b, a));
00281 }

00283 uint128 CityHash128WithSeed(const char *s, size_t len, uint128 seed) {
00284   if (len < 128) {
00285     return CityMurmur(s, len, seed);
00286   }
00287 
00288   // We expect len >= 128 to be the common case.  Keep 56 bytes of state:
00289   // v, w, x, y, and z.
00290   pair<uint64, uint64> v, w;
00291   uint64 x = Uint128Low64(seed);
00292   uint64 y = Uint128High64(seed);
00293   uint64 z = len * k1;
00294   v.first = Rotate(y ^ k1, 49) * k1 + Fetch64(s);
00295   v.second = Rotate(v.first, 42) * k1 + Fetch64(s + 8);
00296   w.first = Rotate(y + z, 35) * k1 + x;
00297   w.second = Rotate(x + Fetch64(s + 88), 53) * k1;
00298 
00299   // This is the same inner loop as CityHash64(), manually unrolled.
00300   do {
00301     x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
00302     y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
00303     x ^= w.second;
00304     y += v.first + Fetch64(s + 40);
00305     z = Rotate(z + w.first, 33) * k1;
00306     v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
00307     w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
00308     std::swap(z, x);
00309     s += 64;
00310     x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
00311     y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
00312     x ^= w.second;
00313     y += v.first + Fetch64(s + 40);
00314     z = Rotate(z + w.first, 33) * k1;
00315     v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
00316     w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
00317     std::swap(z, x);
00318     s += 64;
00319     len -= 128;
00320   } while (LIKELY(len >= 128));
00321   x += Rotate(v.first + z, 49) * k0;
00322   z += Rotate(w.first, 37) * k0;
00323   // If 0 < len < 128, hash up to 4 chunks of 32 bytes each from the end of s.
00324   for (size_t tail_done = 0; tail_done < len; ) {
00325     tail_done += 32;
00326     y = Rotate(x + y, 42) * k0 + v.second;
00327     w.first += Fetch64(s + len - tail_done + 16);
00328     x = x * k0 + w.first;
00329     z += w.second + Fetch64(s + len - tail_done);
00330     w.second += v.first;
00331     v = WeakHashLen32WithSeeds(s + len - tail_done, v.first + z, v.second);
00332   }
00333   // At this point our 56 bytes of state should contain more than
00334   // enough information for a strong 128-bit hash.  We use two
00335   // different 56-byte-to-8-byte hashes to get a 16-byte final result.
00336   x = HashLen16(x, v.first);
00337   y = HashLen16(y + z, w.first);
00338   return uint128(HashLen16(x + v.second, w.second) + y,
00339                  HashLen16(x + w.second, y + v.second));
00340 }

00362 // Requires len >= 240.
00363 static void CityHashCrc256Long(const char *s, size_t len,
00364                                uint32 seed, uint64 *result) {
00365   uint64 a = Fetch64(s + 56) + k0;
00366   uint64 b = Fetch64(s + 96) + k0;
00367   uint64 c = result[0] = HashLen16(b, len);
00368   uint64 d = result[1] = Fetch64(s + 120) * k0 + len;
00369   uint64 e = Fetch64(s + 184) + seed;
00370   uint64 f = seed;
00371   uint64 g = 0;
00372   uint64 h = 0;
00373   uint64 i = 0;
00374   uint64 j = 0;
00375   uint64 t = c + d;
00376 
00377   // 240 bytes of input per iter.
00378   size_t iters = len / 240;
00379   len -= iters * 240;
00380   do {
00381 #define CHUNK(multiplier, z)                                    \
00382     {                                                           \
00383       uint64 old_a = a;                                         \
00384       a = Rotate(b, 41 ^ z) * multiplier + Fetch64(s);          \
00385       b = Rotate(c, 27 ^ z) * multiplier + Fetch64(s + 8);      \
00386       c = Rotate(d, 41 ^ z) * multiplier + Fetch64(s + 16);     \
00387       d = Rotate(e, 33 ^ z) * multiplier + Fetch64(s + 24);     \
00388       e = Rotate(t, 25 ^ z) * multiplier + Fetch64(s + 32);     \
00389       t = old_a;                                                \
00390     }                                                           \
00391     f = _mm_crc32_u64(f, a);                                    \
00392     g = _mm_crc32_u64(g, b);                                    \
00393     h = _mm_crc32_u64(h, c);                                    \
00394     i = _mm_crc32_u64(i, d);                                    \
00395     j = _mm_crc32_u64(j, e);                                    \
00396     s += 40
00397 
00398     CHUNK(1, 1); CHUNK(k0, 0);
00399     CHUNK(1, 1); CHUNK(k0, 0);
00400     CHUNK(1, 1); CHUNK(k0, 0);
00401   } while (--iters > 0);
00402 
00403   while (len >= 40) {
00404     CHUNK(k0, 0);
00405     len -= 40;
00406   }
00407   if (len > 0) {
00408     s = s + len - 40;
00409     CHUNK(k0, 0);
00410   }
00411   j += i << 32;
00412   a = HashLen16(a, j);
00413   h += g << 32;
00414   b += h;
00415   c = HashLen16(c, f) + i;
00416   d = HashLen16(d, e + result[0]);
00417   j += e;
00418   i += HashLen16(h, t);
00419   e = HashLen16(a, d) + j;
00420   f = HashLen16(b, c) + a;
00421   g = HashLen16(j, i) + c;
00422   result[0] = e + f + g + h;
00423   a = ShiftMix((a + g) * k0) * k0 + b;
00424   result[1] += a + result[0];
00425   a = ShiftMix(a * k0) * k0 + c;
00426   result[2] = a + result[1];
00427   a = ShiftMix((a + e) * k0) * k0;
00428   result[3] = a + result[2];
00429 }
