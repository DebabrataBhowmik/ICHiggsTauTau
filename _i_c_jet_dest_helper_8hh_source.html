238 void FillJPTTrackVars(ic::JPTJet *dest,
239  std::vector<reco::Track const *> const &all_tracks,
240  edm::Handle<reco::TrackCollection> const &trk_handle,
241  edm::Handle<reco::VertexCollection> const &vtx_handle,
242  std::map<unsigned, unsigned> const &trk_vtx_map) {
243  std::vector<double> pt_at_vtx_vec(vtx_handle->size(), 0.0);
244  reco::Track const *ptr_first = &(trk_handle->at(0));
245  double trk_pt_total = 0.0;
246  for (unsigned j = 0; j < all_tracks.size(); ++j) {
247  unsigned idx = unsigned(all_tracks[j] - ptr_first);
248  trk_pt_total += all_tracks[j]->pt();
249  // Is track associated to a vertex?
250  if (trk_vtx_map.count(idx) > 0) {
251  pt_at_vtx_vec[trk_vtx_map.find(idx)->second] += all_tracks[j]->pt();
252  } else { // No, so is it within 0.2 cm wrt the closest vertex in z
253  std::vector<double> dz_with_vtx(vtx_handle->size(), 0.0);
254  for (unsigned k = 0; k < vtx_handle->size(); ++k) {
255  dz_with_vtx[k] = fabs(vtx_handle->at(k).z() - all_tracks[j]->vz());
256  }
257  std::vector<double>::const_iterator min =
258  std::min_element(dz_with_vtx.begin(), dz_with_vtx.end());
259  if (min != dz_with_vtx.end()) {
260  if (*min < 0.2) {
261  pt_at_vtx_vec[unsigned(min - dz_with_vtx.begin())] +=
262  all_tracks[j]->pt();
263  }
264  }
265  }
266  }
267  if (vtx_handle->size() > 0 && trk_pt_total > 0.0) {
268  for (unsigned j = 0; j < pt_at_vtx_vec.size(); ++j) {
269  pt_at_vtx_vec[j] = pt_at_vtx_vec[j] / trk_pt_total;
270  }
271  dest->set_beta(pt_at_vtx_vec[0]);
272  dest->set_beta_max(
273  *std::max_element(pt_at_vtx_vec.begin(), pt_at_vtx_vec.end()));
274  dest->set_track_pt_total(trk_pt_total);
275  } else {
276  dest->set_beta(-1.);
277  dest->set_beta_max(-1.);
278  dest->set_track_pt_total(0);
279  }
280 }

291  reco::Track const* ptr_first = &(trk_handle->at(0));
292  for (unsigned iP = 0; iP < pfcands.size(); ++iP) {
293  if (pfcands[iP]->trackRef().isNonnull()) {
294  unsigned idx = unsigned(&(*(pfcands[iP]->trackRef())) - ptr_first);
295  trk_pt_total += pfcands[iP]->trackRef()->pt();
296  // Is track associated to a vertex?
297  if (trk_vtx_map.count(idx) > 0) {

301  ++charged_multiplicity_nopu;

304  pfcands[iP]->rapidity(),
305  pfcands[iP]->phi());
306  }
307  pt_at_vtx_vec[trk_vtx_map.find(idx)->second] +=
308  pfcands[iP]->trackRef()->pt();
309  } else { // No, so is it within 0.2 cm wrt the closest vertex in z
310  std::vector<double> dz_with_vtx(vtx_handle->size(), 0.0);
311  for (unsigned iV = 0; iV < vtx_handle->size(); ++iV) {
312  dz_with_vtx[iV] =
313  fabs(vtx_handle->at(iV).z() - pfcands[iP]->trackRef()->vz());
314  }
315  std::vector<double>::const_iterator min =
316  std::min_element(dz_with_vtx.begin(), dz_with_vtx.end());
317  if (min != dz_with_vtx.end()) {
318  if (*min < 0.2) {
319  // As above, if this track is not associated to any vertex, but
320  // is within 0.2 cm of first PV, then increase multiplicity
321  // count
322  if (unsigned(min-dz_with_vtx.begin()) == 0) {
323  ++charged_multiplicity_nopu;
324  linear_radial_moment_nopu +=
325  pfcands[iP]->pt() * deltaR(dest->vector().Rapidity(), dest->phi(),
326  pfcands[iP]->rapidity(),
327  pfcands[iP]->phi());
328  }
329  pt_at_vtx_vec[unsigned(min - dz_with_vtx.begin())] +=

331  }

333  }

344  dest->set_charged_multiplicity_nopu(charged_multiplicity_nopu);

348  pt_at_vtx_vec[iV] = pt_at_vtx_vec[iV] / trk_pt_total;
349  }
