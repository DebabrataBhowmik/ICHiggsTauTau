00402       nus->set_asymm(valmap.IsAsymm());
00403       nus->set_value_u(valmap.ValU(procs_[i].get()));
00404       nus->set_value_d(valmap.ValD(procs_[i].get()));

00406       nus->set_asymm(true);
00407       nus->set_value_u(1.0);
