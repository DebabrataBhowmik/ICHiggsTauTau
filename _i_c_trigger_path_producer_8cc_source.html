57 void ICTriggerPathProducer::produce(edm::Event& event,
58  const edm::EventSetup& setup) {
59  paths_->clear();
60 
61  if (!input_is_standalone_) {
62  edm::Handle<pat::TriggerEvent> trig_handle;
63  event.getByLabel(input_, trig_handle);
64  std::vector<pat::TriggerPath> const* paths = trig_handle->paths();
65  paths_->reserve(paths->size());
66  for (unsigned i = 0; i < paths->size(); ++i) {
67  pat::TriggerPath const& src = paths->at(i);
68  if (!src.wasAccept() && include_if_fired_) continue;
69  paths_->push_back(ic::TriggerPath());
70  ic::TriggerPath & dest = paths_->back();
71  dest.set_accept(src.wasAccept());
72  dest.set_prescale(src.prescale());
73  std::string name = src.name();
74  SetNameInfo(name, &dest);
75  }
76  } else { // i.e. MiniAOD
77  edm::Handle<edm::TriggerResults> trigres_handle;
78  event.getByLabel(input_, trigres_handle);
79 
80 #if CMSSW_MAJOR_VERSION >= 7
81  edm::Handle<pat::PackedTriggerPrescales> prescales_handle;
82  event.getByLabel(input_prescales_, prescales_handle);
83 #endif
84 
85  edm::TriggerNames const& names = event.triggerNames(*trigres_handle);
86  paths_->reserve(trigres_handle->size());
87  for (unsigned int i = 0, n = trigres_handle->size(); i < n; ++i) {
88  if (!trigres_handle->accept(i) && include_if_fired_) continue;
89  paths_->push_back(ic::TriggerPath());
90  ic::TriggerPath & dest = paths_->back();
91  dest.set_accept(trigres_handle->accept(i));
92 #if CMSSW_MAJOR_VERSION >= 7
93  dest.set_prescale(prescales_handle->getPrescaleForIndex(i));
94 #else
95  dest.set_prescale(0);
96 #endif
97  std::string name = names.triggerName(i);
98  SetNameInfo(name, &dest);
99  }
