00043 double CombineHarvester::GetUncertainty() {

00049     double rate_d = this->GetRateInternal(lookup, param_it.first);
00050     param_it.second->set_val(backup+param_it.second->err_u());
00051     double rate_u = this->GetRateInternal(lookup, param_it.first);

00134     std::string const& single_nus) {
00135   double rate = 0.0;
00136   // TH1F tot_shape
00137   for (unsigned i = 0; i < procs_.size(); ++i) {
00138     double p_rate = procs_[i]->rate();
00139     // If we are evaluating the effect of a single parameter
00140     // check the list of associated nuisances and skip if
00141     // this "single_nus" is not in the list
00142     // However - we can't skip if the process has a pdf, as
00143     // we haven't checked what the parameters are
00144     if (single_nus != "" && !procs_[i]->pdf()) {
00145       if (!ch::any_of(lookup[i], [&](Nuisance const* nus) {
00146         return nus->name() == single_nus;
00147       })) continue;
00148     }
00149     for (auto nus_it : lookup[i]) {
00150       double x = params_[nus_it->name()]->val();
00151       if (nus_it->asymm()) {
00152         if (x >= 0) {
00153           p_rate *= std::pow(nus_it->value_u(), x * nus_it->scale());
00154         } else {
00155           p_rate *= std::pow(nus_it->value_d(), -1.0 * x * nus_it->scale());
00156         }
00157       } else {
00158         p_rate *= std::pow(nus_it->value_u(), x * nus_it->scale());
00159       }
00160     }
00161     rate += p_rate;
00162   }
00163   return rate;

00295 void CombineHarvester::ShapeDiff(double x,
00296     TH1F * target,
00297     TH1 const* nom,
00298     TH1 const* low,
00299     TH1 const* high) {
00300   //      * we already have computed the histogram for diff=(dhi-dlo) and sum=(dhi+dlo)
00301   //      * so we just do template += (0.5 * x) * (diff + smoothStepFunc(x) * sum)
00302   //
00303   double fx = smoothStepFunc(x);
00304   for (int i = 1; i <= target->GetNbinsX(); ++i) {
00305     float h = high->GetBinContent(i);
00306     float l = low->GetBinContent(i);
00307     float n = nom->GetBinContent(i);
00308     target->SetBinContent(i, target->GetBinContent(i) +
00309                                  0.5 * x * ((h - l) + (h + l - 2. * n) * fx));
00310   }
00311 }
