00029 // Extract info from filename using parse rule like:
00030 // ".*{MASS}/{ANALYSIS}_{CHANNEL}_{BINID}_{ERA}.txt"
00031 int CombineHarvester::ParseDatacard(std::string const& filename,
00032     std::string parse_rules) {
00033   boost::replace_all(parse_rules, "$ANALYSIS",  "(?<ANALYSIS>\\w+)");
00034   boost::replace_all(parse_rules, "$ERA",       "(?<ERA>\\w+)");
00035   boost::replace_all(parse_rules, "$CHANNEL",   "(?<CHANNEL>\\w+)");
00036   boost::replace_all(parse_rules, "$BINID",     "(?<BINID>\\w+)");
00037   boost::replace_all(parse_rules, "$MASS",      "(?<MASS>\\w+)");
00038   boost::regex rgx(parse_rules);
00039   boost::smatch matches;
00040   boost::regex_search(filename, matches, rgx);
00041   this->ParseDatacard(filename,
00042     matches.str("ANALYSIS"),
00043     matches.str("ERA"),
00044     matches.str("CHANNEL"),
00045     matches.str("BINID").length() ?
00046       boost::lexical_cast<int>(matches.str("BINID")) : 0,
00047     matches.str("MASS"));
00048   return 0;
00049 }
